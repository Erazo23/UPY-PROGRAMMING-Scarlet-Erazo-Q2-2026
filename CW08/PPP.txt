# INPUT SECTION
PRINT "1. x^2-4x+3 | 2. 3x^3-5x^2+2x-1 | 3. sin(x) | 4. cos(x) | 5. e^x | 6. ln(x)"
READ opc
PRINT "Enter interval limits a and b:"
READ a
READ b
PRINT "1. Default | 2. Custom | 3. Auto-adjust"
READ modo

n ← 100
met ← "MIDPOINT"
umbral ← 0.001

IF modo IS 2 THEN
    READ n
    READ met
ELSE IF modo IS 3 THEN
    READ umbral
    n ← 10
    met ← "MIDPOINT"
ENDIF

# PROCESS SECTION
# Calculate exact analytical integral value
IF opc IS 1 THEN
    v_exacto ← ((b^3)/3 - 2*(b^2) + 3*b) - ((a^3)/3 - 2*(a^2) + 3*a)
# (Repeat analytical structure for options 2 to 6 horizontally or simplified)
ELSE
    v_exacto ← Integration_Formula(opc, a, b)
ENDIF

LOOP
    h ← (b - a) / n
    area ← 0.0
    
    IF met IS "LRM" THEN
        FOR i FROM 0 TO n - 1 DO
            area ← area + f(a + i * h) * h
        ENDFOR
    ELSE IF met IS "RRM" THEN
        FOR i FROM 1 TO n DO
            area ← area + f(a + i * h) * h
        ENDFOR
    ELSE IF met IS "MIDPOINT" THEN
        FOR i FROM 0 TO n - 1 DO
            area ← area + f(a + i * h + h / 2) * h
        ENDFOR
    ELSE IF met IS "TRAPEZOID" THEN
        area ← area + ((f(a) + f(b)) / 2.0) * h
        FOR i FROM 1 TO n - 1 DO
            area ← area + f(a + i * h) * h
        ENDFOR
    ENDIF
    
    err_abs ← ABS(v_exacto - area)
    err_rel ← err_abs / ABS(v_exacto)
    
    IF modo IS 3 AND err_rel > umbral THEN
        n ← n * 2
    ELSE
        EXIT LOOP
    ENDIF
ENDLOOP

# OUTPUT SECTION
PRINT "FINAL RESULTS"
PRINT a, b, n, met
PRINT v_exacto, area
PRINT err_abs, err_rel
